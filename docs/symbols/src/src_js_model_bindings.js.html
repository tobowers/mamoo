1 (function () {
  2     var Binder = function (modelObj, modelKey, targetObj, targetAttribute, opts) {
  3         opts = opts || {};
  4         this.targetObj = targetObj;
  5         this.modelObj = modelObj;
  6         this.modelKey = modelKey;
  7         this.targetAttribute = targetAttribute;
  8         this.opts = opts;
  9         this.eventKey = this.modelKey + "_changed";
 10         this.handleChange = _(this.handleChange).bind(this);
 11         return this;
 12     };
 13 
 14     //handler receives (payload, targetObj, targetAttribute, model, key);
 15     Binder.prototype.handleChange = function (payload) {
 16         var data = this.modelObj.get(this.modelKey);
 17 //        console.log("handling change", data);
 18         if (this.opts.preProcess) {
 19             data = this.opts.preProcess(data);
 20         }
 21         if (this.opts.handler) {
 22             this.opts.handler(data, this.targetObj, this.targetAttribute, this.modelObj, this.modelKey);
 23         } else {
 24             this.targetObj[this.targetAttribute] = data;
 25         }
 26     };
 27 
 28     Binder.prototype.listen = function () {
 29 //        console.log("listening to", this.eventKey);
 30         this.modelObj.on(this.eventKey, this.handleChange);
 31         return this;
 32     };
 33 
 34     Binder.prototype.stopUpdating = function () {
 35         this.modelObj.removeListener(this.eventKey, this.handleChange);
 36     };
 37 
 38 
 39 
 40     var updaterFunc = function (key, opts) {
 41             var binder = new Binder(this, key, opts.object, opts.attribute, opts);
 42             binder.listen();
 43             return binder;
 44     };
 45 
 46     MBX.JsModel.extend({
 47         updatesOn: updaterFunc
 48     });
 49 
 50     MBX.JsModel.extendInstancePrototype({
 51         updatesOn: updaterFunc
 52     })
 53 
 54 })();
