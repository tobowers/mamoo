  8 /** this is the view part of the MVC framework
  9     handle updates and creations and deletes of registered
 10     objects and classes
 11     @namespace
 12 */
 13 MBX.JsView = (function () {
 14     /**
 15         @memberof MBX.JsView
 16         @namespace
 17     */
 18     var self = {};
 19     
 20     var jsElementClass = '.js_updateable';
 21     
 22     /**
 23         Grab give a model - get a css... pass in "." as the second arg to give it css-style return
 24         @param {JsModel} model the model you want the css for
 25         @param {String} prePend any string you want prepended to the return (to avoid "." + MBX.JsView.modelCSS(model))
 26         @returns {String} a css class for a model
 27         @name MBX.JsView.modelCSS
 28         @function
 29     */
 30     self.modelCSS = function (model, prePend) {
 31         prePend = prePend || "";
 32         return prePend + model.modelName.toLowerCase();
 33     };
 34     
 35     /** given an instance of a model, give the recommended instance class 
 36         @param {JsModel#instance} instance the JsModel instance
 37         @param {String} prePend any string you want prepended to the return (to avoid "." + MBX.JsView.modelCSS(model))
 38         @returns {String} a long-ish css class for a specific instance
 39         @name MBX.JsView.cssForInstance
 40         @function
 41     */
 42     self.cssForInstance = function (instance, prePend) {
 43         prePend = prePend || "";
 44         return prePend + self.modelCSS(instance.parentClass) + "_" + instance.primaryKey().toLowerCase().gsub(/[^\w\-]/, "_");
 45     };
 46     
 47     /** @class A single view instance
 48         @constructor
 49         @name JsView
 50         @example
 51             MBX.JsView.create({
 52                 model: MBX.DesktopUpload,
 53                 onCreate: function (upload) {
 54                     //create the upload
 55                 },
 56                 onChange: function (upload) {
 57                     // any upload changes
 58                 },
 59                 onDestroy: function (upload) {
 60                     // handle destroys
 61                 }
 62             });
 63     */
 64     var View = function (opts) {
 65         opts = opts || {};
 66         Object.extend(this, opts);
 67         
 68         if (this.model) {
 69             this._subscribeToEvents();
 70         }
 71         
 72         if (typeof this.afterCreate == 'function') {
 73             this.afterCreate();
 74         }
 75     };
 76     
 77     View.prototype = /** @lends JsView */{
 78         
 79         /** helper function to spit out divs
 80             @param {String} className the classes you want to add to the div
 81             @param {Object} opts (optional) any opts to pass to new Element
 82             @returns div DOM element
 83         */
 84         div: function (className, opts) {
 85             opts = opts || {};
 86             Object.extend(opts, { className: className });
 87             return new Element('div', opts);
 88         },
 89         
 90         /** helper function to spit out imageTag
 91             @param {String} src the url for the image
 92             @param {Object} opts (optional) any opts to pass to new Element
 93             @returns img DOM element
 94         */
 95         imageTag: function (src, opts) {
 96             opts = opts || {};
 97             Object.extend(opts, { src: src });
 98             return new Element("img", opts);
 99         },
100         
101         /** Anytime an instance changes on the model you are observing,
102             JsView will fire a function with the object and the key
103             @params {JsModel#instance} object the instance that changed
104             @params {String} key the key that changed
105         */
106         _onInstanceChange: function (evt) {
107             if (typeof this.onInstanceChange == 'function') {
108                 this.onInstanceChange(evt.object, evt.key);
109             }
110         },
111         
112         _onInstanceCreate: function (evt) {
113             if (typeof this.onInstanceCreate == 'function') {
114                 this.onInstanceCreate(evt.object);
115             }
116         },
117         
118         _onInstanceDestroy: function (evt) {
119             if (typeof this.onInstanceDestroy == 'function') {
120                 this.onInstanceDestroy(evt.object);
121             }
122         },
123         
124         _onAttributeChange: function (evt) {
125             if (typeof this.onAttributeChange == 'function') {
126                 this.onAttributeChange(evt.key);
127             }
128         },
129         
130         /** subscribe to change, create, destroy events
131             We assume you don't need your UI elements to update at the expense of user interaction
132             hence the defer: true
133         */
134         _subscribeToEvents: function () {
135             var changeEvent = this.model.Event.changeInstance;
136             var newEvent = this.model.Event.newInstance;
137             var destroyEvent = this.model.Event.destroyInstance;
138             var attributeEvent = this.model.Event.changeAttribute;
139 
140             this.eventSubscriptions = [];
141             this.eventSubscriptions.push(MBX.EventHandler.subscribe(MBX, changeEvent, this._onInstanceChange.bind(this)));
142             this.eventSubscriptions.push(MBX.EventHandler.subscribe(MBX, newEvent, this._onInstanceCreate.bind(this)));
143             this.eventSubscriptions.push(MBX.EventHandler.subscribe(MBX, destroyEvent, this._onInstanceDestroy.bind(this)));
144             this.eventSubscriptions.push(MBX.EventHandler.subscribe(MBX, attributeEvent, this._onAttributeChange.bind(this)));
145         },
146         
147         /** return the CSS class for the model associated with this view
148             @param {String} prePend any string to prepend to the model
149             @see MBX.JsView.modelCSS
150         */
151         modelCSS: function (prePend) {
152             return MBX.JsView.modelCSS(this.model, prePend);
153         },
154         
155         /** given an instance of a model, give the recommended instance class
156             @see MBX.JsView.cssForInstance
157         */
158         cssForInstance: function (instance, prePend) {
159             return MBX.JsView.cssForInstance(instance, prePend);
160         },
161         
162         /** given an instance, return the relevent elements
163             @param {JsModel#instance} instance the instance you want to find elements
164             @returns {Array} an array of extended dom elements that match the relevent class names
165         */
166         elementsFromInstance: function (instance) {
167             return $$(this.modelCSS(".") + " " + cssForInstance("."));
168         },
169         
170         /** return all the various collections of dom elements that have the appropriate classes
171             @returns {Array} array of extended dom elements that match collection CSS
172         */
173         domCollections: function () {
174             var findString = [jsElementClass, this.modelCSS("."), '.collection'].join('');
175             return $$(findString);
176         }
177         
178     };
179     
180     /** create a new view handler... specify a model and some
181         functions and some great magic happens
182         @name MBX.JsView.create
183         @function
184         @param {Object} opts the various options specified for a view
185         @example
186             MBX.JsView.create({
187                 model: MBX.DesktopUpload,
188                 onCreate: function (upload) {
189                     //create the upload
190                 },
191                 onChange: function (upload) {
192                     // any upload changes
193                 },
194                 onDestroy: function (upload) {
195                     // handle destroys
196                 }
197             });
198     */
199     self.create = function (opts) {
200         return new View(opts);
201     };
202     
203     /**
204         call extend() to add methods and/or attributes to ALL views
205         @param {Object} methsAndAttrs
206         @name MBX.JsView.extend
207         @function
208     */
209     self.extend = function (methsAndAttrs) {
210         methsAndAttrs = methsAndAttrs || {};
211         Object.extend(View.prototype, methsAndAttrs);
212     };
213     
214     /** Added to prototype elements when a view is created
215         allows you to observe particular keys of a model object
216         @name MBX.JsView.updatesOn
217         @function
218         @param {JsModel#instance} obj the model instance to watch
219         @param {String} key the key to watch
220         @param {Object} opts handler or preProcess functions to execute
221         @returns {DOM element} the dom element that was called (for chaining)
222     */
223     self.updatesOn = function (element, obj, key, opts) {
224         if (!obj || !key) {
225             throw new Error("You must specify a key or an object with updatesOn");
226         }
227         element = $(element);
228         opts = opts || {};
229         
230         var changeHandler = function (evt) {
231             var content = obj.get(key);
232             if (opts.preProcess) {
233                 content = opts.preProcess(obj, content, key);
234             }
235             if (opts.handler) {
236                 opts.handler(obj, element, content);
237             } else {
238                 element.update(content);
239             }
240         };
241         
242         var sub = MBX.EventHandler.subscribe(obj, key + "_changed", changeHandler);
243         
244         element.__JsViewSubscriptions = element.__JsViewSubscriptions || [];       
245         element.__JsViewSubscriptions.push(sub);
246         
247         obj.__JsViewSubscriptions = obj.__JsViewSubscriptions || [];
248         obj.__JsViewSubscriptions.push(sub);
249         
250         if (opts.updateNow) {
251             changeHandler();
252         }
253         
254         return element;
255     };
256     
257     /** stop updating this instance, all event handlers are removed
258         Preferred method of accessing this function is through extended elements
259         @name MBX.JsView.stopUpdating
260         @function
261         @param {DOM Element} element the element to stop observing - this will usually be filled for you by prototype
262         @returns {DOM Element} element for chaining
263         @example
264           $("el").stopUpdating();
265     */
266     self.stopUpdating = function (element) {
267         element = $(element);
268         if (element.__JsViewSubscriptions) {
269             while (element.__JsViewSubscriptions.length > 0) {
270                 MBX.EventHandler.unsubscribe(element.__JsViewSubscriptions.pop());
271             }
272         }
273         return element;
274     };
275     
276     /** attach a JsModel#instance to this DOM element
277         Preferred method of accessing this function is through extended elements
278         @name MBX.JsView.assignInstance
279         @function
280         @example
281           $("el").assignInstance(instance);
282           $("el").getInstance();  // == instance
283     */
284     self.assignInstance = function (element, instance) {
285         element = $(element);
286         element.__JsViewMvcInstance = instance;
287         element.addClassName(MBX.JsView.modelCSS(instance.parentClass));
288         element.addClassName(MBX.JsView.cssForInstance(instance));
289         return element;
290     };
291     
292     /** fetch a JsModel#instance from this DOM element
293         Preferred method of accessing this function is through extended elements
294         @name MBX.JsView.getInstance
295         @function
296         @example
297           $("el").assignInstance(instance);
298           $("el").getInstance();  // == instance
299     */
300     self.getInstance = function (element, instance) {
301         return element.__JsViewMvcInstance;
302     };
303     
304     Element.addMethods({
305         updatesOn: self.updatesOn,
306         stopUpdating: self.stopUpdating,
307         assignInstance: self.assignInstance,
308         getInstance: self.getInstance
309     });
310     
311     return self;
312 })();
