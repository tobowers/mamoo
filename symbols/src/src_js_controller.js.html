  8 /**
  9     Create and extend controllers
 10     @class
 11 */
 12 MBX.JsController = (function () {
 13     /**
 14         public methods of JsController
 15         @memberof MBX.JsController
 16         @namespace
 17     */
 18     var publicObj = {};
 19     
 20     /**
 21         used to cache instances of controllers and stop name collisions
 22         @memberof MBX.JsController
 23         @private
 24     */
 25     var controllerCache = {};
 26     
 27     
 28     /** @private */
 29     var jsElementClass = '.js_updateable';
 30     
 31     /**
 32         The instance that gets created from calling
 33         MBX.JsController.create(name, opts)
 34         @see MBX.JsController
 35         @name JsController
 36         @class
 37     */
 38     JsController = function (name, opts) {
 39         opts = opts || {};
 40         if (!name) {
 41             throw new Error("A name must be specified");
 42         }
 43         if (controllerCache[name]) {
 44             throw new Error("A controller by that name exists");
 45         }
 46         
 47         this.controllerName = name;
 48         Object.extend(this, opts);
 49         if (this.model) {
 50             this._subscribeToEvents();
 51         }
 52         controllerCache[name] = this;
 53         
 54         MBX.EventHandler.fireCustom(MBX, publicObj.Event.newController, {
 55             object: this
 56         });
 57     };
 58     
 59     JsController.prototype = 
 60         /** @lends JsController */
 61         {
 62         /**
 63             If you have passed in a function for onInstanceChange
 64             then this will pass the object and the key that changed to
 65             your function
 66             @requires MBX.JsModel
 67             @requires this.model
 68             @see JsModel#instance
 69             @example
 70               myController = MBX.JsController.create("myController", {
 71                   onInstanceChange: function (modelInstance, key) {
 72                       // where key will be the attribute (string)
 73                       // that has changed on the modelInstance (JsModel#instance)
 74                   }
 75               });
 76         */
 77         _onInstanceChange: function (evt) {
 78             if (this.onInstanceChange) {
 79                 this.onInstanceChange(evt.object, evt.key);
 80             }
 81         },
 82         
 83         /**
 84             @private
 85             @requires MBX.JsModel
 86             @requires this.model
 87         */
 88         _onInstanceCreate: function (evt) {            
 89             if (this.onInstanceCreate) {
 90                 this.onInstanceCreate(evt.object);
 91             }     
 92         },
 93         
 94         fireAfterRender: function () {
 95             MBX.EventHandler.fireCustom(MBX, this.controllerName + "_" + MBX.JsController.Event.afterRender);
 96         },
 97         
 98         /**
 99             @private
100             @requires MBX.JsModel
101             @requires this.model
102         */
103         _onInstanceDestroy: function (evt) {
104             if (this.onInstanceDestroy) {
105                 this.onInstanceDestroy(evt.object);
106             }
107         },
108         
109         /**
110             @private
111             @requires MBX.JsModel
112             @requires this.model
113         */
114         _onAttributeChange: function (evt) {
115             if (typeof this.onAttributeChange == 'function') {
116                 this.onAttributeChange(evt.key);
117             }
118         },
119         
120         /**
121             @private
122             @requires MBX.JsModel
123             @requires this.model
124             
125         */
126         _subscribeToEvents: function () {
127             var model = this.model;
128             if (model.constructor != Array) {
129                model = [model];
130             }
131             
132             model.each(function (model) {
133                 var changeEvent = model.Event.changeInstance;
134                 var newEvent = model.Event.newInstance;
135                 var destroyEvent = model.Event.destroyInstance;
136                 var attributeEvent = model.Event.changeAttribute;
137                 var defer = this.looselyCoupled;
138 
139                 this.eventSubscriptions = [];
140                 this.eventSubscriptions.push(MBX.EventHandler.subscribe(MBX, changeEvent, this._onInstanceChange.bind(this), {defer: defer}));
141                 this.eventSubscriptions.push(MBX.EventHandler.subscribe(MBX, newEvent, this._onInstanceCreate.bind(this), {defer: defer}));
142                 this.eventSubscriptions.push(MBX.EventHandler.subscribe(MBX, destroyEvent, this._onInstanceDestroy.bind(this), {defer: defer}));
143                 this.eventSubscriptions.push(MBX.EventHandler.subscribe(MBX, attributeEvent, this._onAttributeChange.bind(this), {defer: defer}));
144             }.bind(this));
145         },
146         
147         /**
148             @private
149             @requires MBX.JsModel
150             @requires this.model
151         */
152         _unsubscribeToEvents: function () {
153             if (this.eventSubscriptions && this.eventSubscriptions[0]) {                
154                 this.eventSubscriptions.each(function (subscription) {
155                     MBX.EventHandler.unsubscribe(subscription);
156                 });
157             }
158         }
159     };
160     
161     /**
162         This is mostly used internally and is fired on MBX everytime a controller is created
163         @memberOf MBX.JsController
164         @name MBX.JsController.Event
165     */
166     publicObj.Event = {
167         newController: "new_controller",
168         afterRender: "render_finished"
169     };
170     
171     /**
172         call extend() to add methods and/or attributes to ALL controllers
173         @param {Object} methsAndAttrs
174         @name MBX.JsController.extend
175         @function
176     */
177     publicObj.extend = function (methsAndAttrs) {
178         methsAndAttrs = methsAndAttrs || {};
179         Object.extend(JsController.prototype, methsAndAttrs);
180     };
181     
182     /**
183         Controllers allow some decently powerful hooks. You can specify a model, and an
184         onInstanceChange, onInstanceDestroy, onInstanceCreate.
185         
186         If your controller listens to a model, but you are not dependent on real-time updates,
187         you can add the option "looselyCoupled: true" and all updates will be done with
188         setTimeout, which will be a performance enhancement.
189           
190         @name MBX.JsController.create
191         @param {String} name the name of the controller
192         @param {Object} opts used to extend the controller methods at instantiation
193         @see JsController
194         @function
195         @example
196           MBX.DesktopUploadController = MBX.JsController.create("DesktopUpload", {
197               looselyCoupled: false, // false is the default
198               ANewMethod: function (something) {
199                   return something;
200               }
201           })
202           MBX.DesktopUpload.ANewMethod("boo") == "boo";
203         @example
204           MBX.DesktopUploadController = MBX.JsController.create("DesktopUpload", {
205               model: MBX.DesktopUpload,
206               onInstanceCreate: function (instance) {
207                   alert(instance.get('greeting'));
208               }              
209           });
210           MBX.DesktopUpload.create({ greeting: 'hi' });  // will alert('hi');
211         @example
212           MBX.DesktopUploadController = MBX.JsController.create("DesktopUpload", {
213               model: MBX.DesktopUpload,
214               onInstanceChange: function (instance) {
215                   alert(instance.get('greeting'));
216               }              
217           });
218           var instance = MBX.DesktopUpload.create();
219           instance.set('greeting', 'hi');  // will alert('hi')
220     */
221     publicObj.create = function (name, opts) {
222         if (controllerCache[name]) {
223             throw new Error("A controller with the name of " + name + " is already in use");
224         }
225         var controller = new JsController(name, opts);
226         if (typeof controller.initialize == "function") {
227             controller.initialize();
228         }
229         return controller;
230     };
231     
232     /**
233         Destroy a controller and unsubscribe its event listeners
234         @param {String} name the name of the controller
235         @name MBX.JsController.destroyController
236         @function
237     */
238     publicObj.destroyController = function (name) {
239         if (controllerCache[name]) {
240             controllerCache[name]._unsubscribeToEvents();
241             delete controllerCache[name];
242         }
243     };
244     
245     return publicObj;
246 })();
